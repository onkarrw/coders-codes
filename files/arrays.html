public class Arrays {

    

    public static void main(String[] args) {

       //jagged arrays

       

       

       int arr[][]  = {

              {3,6,8,9},

              {1,5,8},

              {7,4,2,1,5}          

       };

       

       //finding row length

       for (int i=0; i<3; i++)

       {

           System.out.println(arr[i].length);

           

       }

       

       //printing jagged array

       

       for (int i=0; i<3; i++) 

       {

           for (int j=0; j<arr[i].length; j++)

           {

               System.out.print(arr[i][j]+" ");

              

           }

           System.out.println();

           

       }

              

       }

}
